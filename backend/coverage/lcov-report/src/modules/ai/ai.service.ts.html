import { Injectable } from '@nestjs/common';
import Anthropic from '@anthropic-ai/sdk';
import { GoogleGenerativeAI, Content } from '@google/generative-ai';
import { Response } from 'express';
import { PrismaService } from '../../prisma/prisma.service';
import { RagService } from './rag.service';
 
@Injectable()
export class AiService {
  private claude = new Anthropic({ apiKey: process.env.ANTHROPIC_API_KEY });
  private gemini = new GoogleGenerativeAI(process.env.GEMINI_API_KEY || '');
 
  constructor(private prisma: PrismaService, private rag: RagService) {}
 
  async streamMessage(conversationId: string, userMessage: string, res: Response): Promise<void> {
    const start = Date.now();
 
    // RAG search
    let context = '';
    let sources: any[] = [];
    try {
      const chunks = await this.rag.search(userMessage, 5);
      context = this.rag.buildContext(chunks);
      sources = chunks.map(c => ({
        id: c.id,
        titre: c.documentTitre,
        extrait: c.texte.slice(0, 120) + '...',
      }));
    } catch {
      // RAG unavailable, continue without context
    }
 
    // Message history
    const history = await this.prisma.message.findMany({
      where: { conversationId },
      orderBy: { createdAt: 'asc' },
      take: 10,
    });
 
    // AI settings
    let settings = await this.prisma.aiSettings.findFirst();
    if (!settings) {
      settings = await this.prisma.aiSettings.create({ data: {} });
    }
 
    const systemPrompt = settings.promptSysteme + (context
      ? `\n\nContexte documentaire pertinent :\n${context}`
      : '');
 
    // SSE headers
    res.setHeader('Content-Type', 'text/event-stream');
    res.setHeader('Cache-Control', 'no-cache');
    res.setHeader('Connection', 'keep-alive');
    res.setHeader('X-Accel-Buffering', 'no');
 
    res.write(`data: ${JSON.stringify({ type: 'sources', sources })}\n\n`);
 
    const messages: Anthropic.MessageParam[] = [
      ...history.map(m => ({
        role: m.sender === 'USER' ? 'user' as const : 'assistant' as const,
        content: m.contenu,
      })),
      { role: 'user', content: userMessage },
    ];
 
    // Try Claude first, fallback to Gemini on failure
    const claudeOk = await this.streamWithClaude(
      settings.modele,
      settings.maxTokens,
      settings.temperature,
      systemPrompt,
      messages,
      conversationId,
      userMessage,
      start,
      res,
      settings,
    );
 
    if (!claudeOk) {
      const geminiModel = settings.modeleSecours || 'gemini-2.0-flash';
      res.write(`data: ${JSON.stringify({ type: 'fallback', model: geminiModel })}\n\n`);
 
      const geminiOk = await this.streamWithGemini(
        geminiModel,
        systemPrompt,
        messages,
        conversationId,
        userMessage,
        start,
        res,
        settings,
      );
 
      if (!geminiOk) {
        res.write(`data: ${JSON.stringify({
          type: 'error',
          message: "Aucune IA disponible pour le moment. Veuillez vérifier vos clés API ou réessayer plus tard.",
        })}\n\n`);
        res.end();
      }
    }
  }
 
  private streamWithClaude(
    model: string,
    maxTokens: number,
    temperature: number,
    systemPrompt: string,
    messages: Anthropic.MessageParam[],
    conversationId: string,
    userMessage: string,
    start: number,
    res: Response,
    settings: any,
  ): Promise<boolean> {
    return new Promise((resolve) => {
      let fullContent = '';
      let tokensInput = 0;
      let tokensOutput = 0;
      let resolved = false;
 
      const finish = (ok: boolean) => {
        if (!resolved) { resolved = true; resolve(ok); }
      };
 
      try {
        const stream = this.claude.messages.stream({
          model,
          max_tokens: maxTokens,
          temperature,
          system: systemPrompt,
          messages,
        });
 
        stream.on('text', (text) => {
          fullContent += text;
          res.write(`data: ${JSON.stringify({ type: 'chunk', content: text, model: 'claude' })}\n\n`);
        });
 
        stream.on('message', async (msg) => {
          tokensInput = msg.usage.input_tokens;
          tokensOutput = msg.usage.output_tokens;
 
          await this.prisma.message.create({
            data: {
              conversationId,
              sender: 'AI',
              contenu: fullContent,
              tokens: tokensOutput,
              modeleUtilise: model,
            },
          });
 
          await this.prisma.aiLog.create({
            data: {
              conversationId,
              prompt: userMessage,
              reponse: fullContent,
              modele: model,
              tokensInput,
              tokensOutput,
              dureeMs: Date.now() - start,
              cout: (tokensInput * 0.000003) + (tokensOutput * 0.000015),
            },
          });
 
          res.write(`data: ${JSON.stringify({ type: 'done', model: 'claude' })}\n\n`);
          res.end();
          finish(true);
        });
 
        stream.on('error', () => {
          // On any Claude error (credits, invalid key, network, etc.) → trigger Gemini fallback
          finish(false);
        });
      } catch (err: any) {
        finish(false);
      }
    });
  }
 
  private async streamWithGemini(
    model: string,
    systemPrompt: string,
    messages: Anthropic.MessageParam[],
    conversationId: string,
    userMessage: string,
    start: number,
    res: Response,
    settings: any,
  ): Promise<boolean> {
    if (!process.env.GEMINI_API_KEY) {
      return false;
    }
 
    try {
      const genModel = this.gemini.getGenerativeModel({
        model,
        systemInstruction: systemPrompt,
      });
 
      // Build Gemini history (exclude last user message)
      const history: Content[] = messages.slice(0, -1).map(m => ({
        role: m.role === 'user' ? 'user' : 'model',
        parts: [{ text: m.content as string }],
      }));
 
      const chat = genModel.startChat({ history });
      const result = await chat.sendMessageStream(userMessage);
 
      let fullContent = '';
      for await (const chunk of result.stream) {
        const text = chunk.text();
        if (text) {
          fullContent += text;
          res.write(`data: ${JSON.stringify({ type: 'chunk', content: text, model: 'gemini' })}\n\n`);
        }
      }
 
      // Save AI message to DB
      await this.prisma.message.create({
        data: {
          conversationId,
          sender: 'AI',
          contenu: fullContent,
          tokens: null,
          modeleUtilise: model,
        },
      });
 
      await this.prisma.aiLog.create({
        data: {
          conversationId,
          prompt: userMessage,
          reponse: fullContent,
          modele: model,
          tokensInput: 0,
          tokensOutput: 0,
          dureeMs: Date.now() - start,
          cout: 0,
        },
      });
 
      res.write(`data: ${JSON.stringify({ type: 'done', model: 'gemini' })}\n\n`);
      res.end();
      return true;
    } catch (err: any) {
      return false;
    }
  }
 
  async generateChecklist(procedureTitle: string, userContext: string): Promise<string[]> {
    const prompt = `Génère une checklist en JSON (tableau de strings) pour la procédure "${procedureTitle}". Contexte : ${userContext}. Réponds uniquement avec le JSON, sans markdown.`;
 
    // Try Claude first
    try {
      const msg = await this.claude.messages.create({
        model: 'claude-sonnet-4-6',
        max_tokens: 512,
        messages: [{ role: 'user', content: prompt }],
      });
      return JSON.parse((msg.content[0] as any).text);
    } catch {
      // Fallback to Gemini
    }
 
    if (process.env.GEMINI_API_KEY) {
      try {
        const model = this.gemini.getGenerativeModel({ model: 'gemini-2.0-flash' });
        const result = await model.generateContent(prompt);
        const text = result.response.text().replace(/```json|```/g, '').trim();
        return JSON.parse(text);
      } catch {}
    }
 
    return [];
  }
 
  async summarizeDocument(text: string): Promise<string> {
    const prompt = `Résume ce document administratif en 3-5 phrases en français :\n\n${text.slice(0, 3000)}`;
 
    // Try Claude first
    try {
      const msg = await this.claude.messages.create({
        model: 'claude-sonnet-4-6',
        max_tokens: 256,
        messages: [{ role: 'user', content: prompt }],
      });
      return (msg.content[0] as any).text;
    } catch {
      // Fallback to Gemini
    }
 
    if (process.env.GEMINI_API_KEY) {
      try {
        const model = this.gemini.getGenerativeModel({ model: 'gemini-2.0-flash' });
        const result = await model.generateContent(prompt);
        return result.response.text();
      } catch {}
    }
 
    return 'Résumé non disponible.';
  }
